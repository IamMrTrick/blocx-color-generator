Quick Start Example

// Step 1: Get color suggestions
const response = await fetch('/api/colors/suggestions', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ primaryColor: '#3B82F6' })
});

const suggestions = await response.json();
console.log(suggestions.data.suggestions);

// Step 2: Generate complete color system
const generateResponse = await fetch('/api/colors/generate', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    primaryColor: '#3B82F6',
    secondaryColor: suggestions.data.suggestions[0].color,
    grayTheme: 'blue',
    backgroundTheme: 'auto'
  })
});

const colorSystem = await generateResponse.json();
const sessionId = colorSystem.data.sessionId;

// Step 3: Export with advanced options
const params = new URLSearchParams({
  sessionId: sessionId,
  prefix: '--my-app',
  caseStyle: 'snake_case',
  mode: 'both',
  structure: 'sections'  // for JSON only
});

// Download CSS with snake_case variables
window.open(`/api/colors/export/css?${params}`);

// Or export JSON with themes structure
params.set('structure', 'themes');
window.open(`/api/colors/export/json?${params}`);
